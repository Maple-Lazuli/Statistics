Part i - Perform backward elimination, create the ROC curve, and find AUC

#Build Intercept Only Model. NOTE: ~ 1 tells R that you only want an intercept
int_only_model <- glm(SurvivedNum ~ 1, 
                      family = binomial,
                      data = titanic)

#Build model with all potential regressors. 
#In code below, SurvivedNum ~ . tells R to use all columns in dataset to predict SurvivedNum
#SurvivedNum ~ . -Survived -Survived2 tells R to use all columns except Survived and Survived2 to predict SurvivedNum. If you get an error saying that object Survived2 cannot be found, delete it from the code below. 
full_model <- glm(SurvivedNum ~ . -Survived -Survived2, 
                  family = binomial,
                  data = titanic)

#Perform backward elimination
#Have R do it all
stats::step(object = full_model, 
            scope = list(lower = int_only_model, upper = full_model),
            data = titanic,
            direction = "backward")
