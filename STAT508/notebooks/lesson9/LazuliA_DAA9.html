conditional_prob <- function(condition, truth, prediction) {
  numerator <- sum((truth == condition) & (prediction == condition))
  denominatior <- sum(truth == condition)
  return(numerator/denominatior)
}

Problem 3 - k-Nearest Neighbors

Part a. - Prepare the data. Using only the quantitative variables in df2 as potential features, scale the features to have mean of 0 and a standard deviation of 1. Then, use a seed of 123 to perform an 80/20 training/validation split.

df2_quant <- df2 %>% select(-Gender, -Education, -HomeOwn, -Diabetes, -Depressed, -PhysActive, -MaritalStatus)
xvars <- c("Age", "Poverty", "HomeRooms", "Weight", "Height", "BMI", "Pulse", "BPSysAve", "BPDiaAve")
df2_quant[, xvars] <- scale(x = df2_quant[, xvars], center = TRUE, scale = TRUE)
set.seed(123)
trainInd <- sample(1:nrow(df2_quant), floor(0.80 * nrow(df2_quant)))
set.seed(NULL)
Train <- df2_quant[trainInd, ]
Validation <- df2_quant[-trainInd, ]

Part b. - Implement kNN classification. Using values of k ranging from 1 to 20, create a plot showing the accuracy values based on predictions on the validation data as a function of k.

num_folds <- 10
folds <- cut(x = 1:nrow(df2_quant), breaks = num_folds, labels = FALSE)

set.seed(123)
folds <- sample(folds)
set.seed(NULL)

maxK <- 20
accMatrix <- matrix(NA, nrow = num_folds, ncol = maxK)

for(i in 1:num_folds){
  valInd <- which(folds == i)
  Validation <- df2_quant[valInd,]
  Train <- df2_quant[-valInd,]
  
  for(j in 1:maxK){
    knnResTemp <- 
    knn(train = Train[ , xvars, drop = FALSE],
        test = Validation[ , xvars, drop = FALSE],
        cl = Train$HealthGen,
        k = j)

    Validation <-
      Validation %>%
      mutate(predHealth = knnResTemp)
    
    pred<- factor(knnResTemp, levels = levels(df2_quant$HealthGen))
    truth <- factor(Validation$HealthGen, levels = levels(df2_quant$HealthGen))
    
    accMatrix[i,j] <- mean(pred == truth)
    }
}

accMeans <- apply(accMatrix, 2, mean)

tempDF <- data.frame(k = 1:maxK, Accuracy = accMeans)

ggplot(data = tempDF,
       mapping = aes(x = k, y = Accuracy)) +
  geom_line() +
  labs(x = "Number of Nearest Neighbors",
       y = "Accuracy") +
  geom_point(data = tempDF[which.max(accMeans), ],
             color = "red", shape = 1, size =3)
